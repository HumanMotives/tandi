const graveListEl = document.getElementById('graveList'); const paginationEl = document.getElementById('pagination'); const perPage = 25; let dataArr = [], current = 1; async function fetchBurials() { const { data, error } = await sb .from('burials') .select('name, method, epitaph, country, timestamp') .order('timestamp', { ascending: false }); if (error) return console.error(error.message); dataArr = data; render(1); } function render(page) { current = page; graveListEl.innerHTML = ''; const start = (page - 1) * perPage; dataArr.slice(start, start + perPage).forEach(b => { const date = new Date(b.timestamp).toLocaleDateString(); const statusIcon = b.method === 'cremate' ? 'icons/icon_pile.png' : 'icons/icon_tombstone.png'; const playIcon = b.method === 'bury' ? `` : ''; const name = b.name.length > 30 ? b.name.slice(0,27) + '...' : b.name; const epitaph = b.epitaph || ''; const countryCode = b.country || ''; const record = document.createElement('div'); record.className = 'burial-record'; record.innerHTML = `
${name}
${date} ${date} ${playIcon}
"${epitaph}"
0 ${countryCode}
`; graveListEl.appendChild(record); }); paginate(); } function paginate() { paginationEl.innerHTML = ''; const total = Math.ceil(dataArr.length / perPage); for (let i=1; i<=total; i++) { const btn = document.createElement('button'); btn.textContent = i; if (i===current) btn.disabled = true; btn.onclick = () => render(i); paginationEl.appendChild(btn); } } document.addEventListener('DOMContentLoaded', fetchBurials);
